#include <stdio.h>
#include <math.h>

int main() {
    int x, n, choice;
    float result;

    printf("Enter value of X: ");
    scanf("%d", &x);

    printf("Enter value of N: ");
    scanf("%d", &n);

    printf("Enter your choice of operation: \n");
    printf("1. Pow(x,n)\n");
    printf("2. Add(x,n)\n");
    printf("3. Sub(x,n)\n");
    printf("4. Mul(x,n)\n");
    printf("5. Div(x,n)\n");

    scanf("%d", &choice);

    switch(choice) {
        case 1:
            result = pow(x, n);
            printf("Pow(%d,%d) = %.2f", x, n, result);
            break;
        case 2:
            result = x + n;
            printf("Add(%d,%d) = %.2f", x, n, result);
            break;
        case 3:
            result = x - n;
            printf("Sub(%d,%d) = %.2f", x, n, result);
            break;
        case 4:
            result = x * n;
            printf("Mul(%d,%d) = %.2f", x, n, result);
            break;
        case 5:
            if(n == 0) {
                printf("Divide by zero error");
            } else {
                result = (float) x / n;
                printf("Div(%d,%d) = %.2f", x, n, result);
            }
            break;
        default:
            printf("Invalid choice");
    }

    return 0;
}
